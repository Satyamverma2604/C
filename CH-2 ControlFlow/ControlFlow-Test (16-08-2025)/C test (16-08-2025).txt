Question 1.


#include<stdio.h>
int main(){

    int n,a,count=0;
    scanf("%d",&n);
    a = n;
   start:
    if(n>0){
        n/=10;
        count ++;
        goto start;
    }
    if(count == 3){
        printf("Previous multiple : %d\n",((a/100))*100);
        printf("next multiple : %d",((a/100)+1)*100);

    }else{
        printf("Invalid Input");
    }

    return 0;
}
