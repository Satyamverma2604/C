Question 1.

#include <stdio.h>

int main()
{
    int a;
    scanf("%d", &a);
    int sum = 0;
    int lastDigit = a %10;
    sum = sum + lastDigit;
    a /= 10;
     lastDigit = a %10;
    sum = sum + lastDigit;
    printf("The sum is %d ", sum);
    return 0;
}
